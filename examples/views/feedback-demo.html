// 모듈 로드
const Feedback = await CATUI.use('feedback');
const { Toast } = await CATUI.use('overlays');

// Toast
Toast.success('성공 메시지');
Toast.error('에러 메시지');

// Notification
const noti = new Feedback.Notification({
  position: 'top-right'
});
noti.show({ title: '알림', message: '내용' });

// ProgressTracker
const progress = new Feedback.ProgressTracker({
  container: '#progress',
  steps: [...],
  currentStep: 1
});

// Skeleton
const skeleton = new Feedback.Skeleton({
  container: '#skeleton',
  type: 'card'
});

// Loading
Feedback.Loading.show({ text: '로딩중...' });
Feedback.Loading.hide();
